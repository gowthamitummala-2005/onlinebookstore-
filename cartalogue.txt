  <script>
    function addToCart(book, price) {
      let cart = JSON.parse(localStorage.getItem("cart")) || [];
      let existing = cart.find(item => item.book === book);
      if (existing) {
        existing.qty += 1;
        existing.amount = existing.qty * existing.price;
      } else {
        cart.push({ book, price: parseFloat(price), qty: 1, amount: parseFloat(price) });
      }
      localStorage.setItem("cart", JSON.stringify(cart));
      alert(book + " added to cart!");
    }
  </script>
